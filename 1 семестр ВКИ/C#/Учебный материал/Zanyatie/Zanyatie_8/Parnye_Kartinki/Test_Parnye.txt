        private void InitializeArray() //Инициализация двуменрного массива pole.
        {
            pole = new int[n, n]; //Инициализация двуменрного массива pole.

            //Заполнение массива нулями.
            for (int i = 0; i < n; i++)
            {
                for (int j = 0; j < n; j++)
                {
                    pole[i, j] = 0;
                }
            }
        }

        private void FillArray_2()
        {
            /*Задача 2. Заполните массив так:
             * в случайные места задайте следующие числа:
             * две единицы, две двойки, …, две восьмерки.
            */

            Random rand = new Random();
            int r = rand.Next(0, n);
            int k = rand.Next(0, n);
           
            int sum = 0;
            pole[r, k] = sum % 8;

            for (int i = 0; i < n; i++)
            {
                for (int j = 0; j < n; j++)
                {
                    sum++;
                    while (pole[r, k] != 0)
                    {
                        r = rand.Next(0, n);
                        k = rand.Next(0, n);
                    }
                    pole[r, k] = sum % 8;
                }
            }
        }
