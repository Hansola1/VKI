using System;
using System.Collections.Generic;
using System.ComponentModel;
using System.Data;
using System.Drawing;
using System.Linq;
using System.Text;
using System.IO;
using System.Threading.Tasks;
using System.Windows.Forms;

namespace RabotaSFailami
{
    public partial class Form1 : Form
    {
        int[] mas = new int[20];
        public Form1()
        {
            InitializeComponent();
        }

        private void button1_Click(object sender, EventArgs e)
        {

            
            if (openFileDialog1.ShowDialog() == DialogResult.OK)
            {               
                int nom = 0;
                for (int ii = 0; ii < 20; ii++)
                {
                    mas[ii] = 0;
                 }
                int t = 0;
                // Объявить экземпляр класса StreamReader
                string filename = openFileDialog1.FileName;
                using (StreamReader fIn = new StreamReader(filename))
                {
                   
                    // Метод Peek() - определить, конец ли файла
                    while (fIn.Peek() != -1)
                    {
                        string s = fIn.ReadLine(); // прочитать строку
                        int len = s.Length;
                       
                        string[] subs = s.Split(' ');
                        int i = 0; 
                        
                        for (int jj = 0; jj < subs.Length; jj++)
                        {
                            mas[nom] = int.Parse(subs[jj]);                      
                            t += mas[nom];
                            nom++;
                        }                       

                        textBox2.Text = s;
                       
                        textBox1.Text =t.ToString();
                        
                        // break;
                    }

                    fIn.Close();
                }
                
            }
        }
    }
}
