&НаСервере
Функция ВойтиНаСервере(Логин, Пароль)
    // Поиск пользователя
    ПользовательСсылка = Справочники.Пользователи.НайтиПоНаименованию(Логин, Истина);
    Если ПользовательСсылка.Пустая() Тогда
        Возврат Новый Структура("Успех", Ложь);
    КонецЕсли;
    
    Пользователь = ПользовательСсылка.ПолучитьОбъект();
    
    // Проверка пароля
    Если Пользователь.Пароль <> Пароль Тогда
        Пользователь.КоличествоПопыток = Пользователь.КоличествоПопыток + 1;
        Пользователь.Записать();
        Возврат Новый Структура("Успех, Причина, Попытки", 
            Ложь, "НеверныйПароль", Пользователь.КоличествоПопыток);
    КонецЕсли;
	
	Если НЕ Пользователь.ДатаВхода = Дата("01.01.0001 0:00:00") Тогда
        Пользователь.Активность = Ложь;  
        Пользователь.Записать();
    КонецЕсли;
    
    // Проверка активности
    Если НЕ Пользователь.Активность Тогда
        Возврат Новый Структура("Успех, Причина", Ложь, "Заблокирован");
    КонецЕсли;
    
    ПервыйВход = (Пользователь.ДатаВхода = Дата("01.01.0001 00:00:00"));
    Пользователь.КоличествоПопыток = 0;
    Пользователь.ДатаВхода = ТекущаяДата();
    Пользователь.Записать();
    
    Возврат Новый Структура(
        "Успех, Роль, ПервыйВход, ПользовательСсылка",
        Истина, Пользователь.Роль, ПервыйВход, ПользовательСсылка);
КонецФункции

&НаКлиенте
Процедура Войти(Команда)
    Результат = ВойтиНаСервере(Логин, Пароль);
    
    Если НЕ Результат.Успех Тогда
        Если Результат.Причина = "НеверныйПароль" Тогда
            Если Результат.Попытки > 3 Тогда
                Сообщить("Вы заблокированы. Обратитесь к администратору");
            Иначе
                Сообщить("Неверный логин или пароль. Проверьте данные");
            КонецЕсли;
        Иначе
            Сообщить("Вы заблокированы. Обратитесь к администратору");
        КонецЕсли;
        Возврат;
    КонецЕсли;
    
    // Успешная авторизация
    Если Результат.ПервыйВход Тогда
        ОткрытьФорму("ОбщаяФорма.ФормаСменыПароля", Новый Структура("ПараметрПользователь", 
		Результат.ПользовательСсылка));
        Сообщить("Измените пароль при первом входе");
    Иначе
        Если Результат.Роль = ПредопределенноеЗначение("Перечисление.Роли.Пользователь") Тогда
            ОткрытьФорму("ОбщаяФорма.ФормаПользователя");
        ИначеЕсли Результат.Роль = ПредопределенноеЗначение("Перечисление.Роли.Администратор") Тогда
            ОткрытьФорму("Справочник.Пользователи.ФормаСписка");
        КонецЕсли;
        Сообщить("Вы успешно авторизовались");
    КонецЕсли; 
    
    ЭтаФорма.Закрыть();
КонецПроцедуры
